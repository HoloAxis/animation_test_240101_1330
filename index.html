import * as THREE from 'three'; import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js'; import { DRACOLoader } from 'three/addons/loaders/DRACOLoader.js'; const scene = new THREE.Scene(); const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); const renderer = new THREE.WebGLRenderer(); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); const loader = new GLTFLoader(); const dracoLoader = new DRACOLoader(); dracoLoader.setDecoderPath('https://www.gstatic.com/draco/versioned/decoders/1.4.1/'); loader.setDRACOLoader(dracoLoader); const clock = new THREE.Clock(); let mixer1, mixer2; // Load the first model loader.load('model1.gltf', function (gltf1) { mixer1 = new THREE.AnimationMixer(gltf1.scene); gltf1.animations.forEach((clip) => { mixer1.clipAction(clip).play(); }); scene.add(gltf1.scene); }); // Load the second model loader.load('model2.gltf', function (gltf2) { mixer2 = new THREE.AnimationMixer(gltf2.scene); gltf2.animations.forEach((clip) => { mixer2.clipAction(clip).play(); }); scene.add(gltf2.scene); }); camera.position.z = 5; function animate() { requestAnimationFrame(animate); const delta = clock.getDelta(); if (mixer1) { mixer1.update(delta); } if (mixer2) { mixer2.update(delta); } renderer.render(scene, camera); } animate();
